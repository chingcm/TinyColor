    getLibraryColor Get color from the library by name. Assumes: color is one of the color in tinycolor.library. Example: “flatui-turquoise”, “flatui-turquoise-900” Returns: tinycolor object.

        function getLibraryColor(color) {
        var match, type, name, subname, result = "color not found";
        color = color.toLowerCase();
        if ((match = matchers.bootstrap.exec(color))) {
            type = match[1];
            if (typeof match[2] === "undefined") type += 4;
            name = match[3];
        } else if ((match = matchers.ncs.exec(color))) {
            type = match[1];
            name = "S " + (match[3] + match[4]).toUpperCase();
        } else if ((match = matchers.flatui.exec(color)) || (match = matchers.material.exec(color))) {
            type = match[1];
            name = match[3];
            subname = typeof match[5] === "undefined" ? 500 : match[5];
        } else if ((match = matchers.general.exec(color))) {
            type = match[1];
            name = match[2];
        }

        if (type in tinycolor.library && typeof tinycolor.library[type] === "object") {
            if (name in tinycolor.library[type]) {
                if (typeof tinycolor.library[type][name] === "string") {
                    result = tinycolor.library[type][name];
                } else if (typeof tinycolor.library[type][name] === "object" && subname in tinycolor.library[type][name]) {
                    result = tinycolor.library[type][name][subname];
                }
            }
        }

        return tinycolor(result);
    }

    toClosestNcs Find the closest NCS value from given data. This function is an alternative to the function in tinycolor.js. It searches the data from tinycolor.library.ncs instead of self-generating a possible color. Since NCS only contains 1950 colors, it is very likely that it cannot find the closest color. See TinyColor Docs for more information

        function toClosestNcs(DATA) {
        var data = JSON.parse(JSON.stringify(DATA)),
            s = data.blackness,
            allBlackness = Object.keys(tinycolor.library.ncs).map(function (v) { return parseInt(v.match(/([0-9]{4})/)[0].slice(0, 2), 10); }).filter(function (v, i, l) { return l.indexOf(v) == i; }).sort(function (a, b) { return Math.abs(s - a) - Math.abs(s - b); }),
            maxS = s + 10 - (s % 10),
            possibleNcs, distance;

        allBlackness.map(function (v) { if (v > s && maxS == s) maxS = v; });
        if (maxS > 90) maxS = 90;

        possibleNcs = Object.keys(tinycolor.library.ncs).filter(function (v) { return parseInt(v.slice(2, 4), 10) <= maxS && parseInt(v.slice(2, 4), 10) >= s - 2; });
        possibleNcs.map(function (code) {
            var color = tinycolor(tinycolor.library.ncs[code]), isClose, smallAngle = function (x1, x2, y1, y2) {
                return (Math.atan(x1 / y1) - Math.atan(x2 / y2)) * 180 / Math.PI < 10;
            };
            distance = color.distance(data.original, true);
            diffR = color._r - data.original.r;
            diffG = color._g - data.original.g;
            diffB = color._b - data.original.b;
            isClose = smallAngle(color._r, data.original.r, color._g, data.original.g) && smallAngle(color._b, data.original.b, color._g, data.original.g) && smallAngle(color._r, data.original.r, color._b, data.original.b);
            if (distance < data.diff && color.isCloseTo(data.original)) {
                data.bestCode = code;
                data.diff = distance;
            }
        });
        if (!(data.hasOwnProperty("bestCode"))) return false;
        data.best = {
            s: parseInt(data.bestCode.slice(2, 4), 10),
            c: parseInt(data.bestCode.slice(4, 6), 10),
            n: data.bestCode.split("-")[1]
        };
        return data;
    }
